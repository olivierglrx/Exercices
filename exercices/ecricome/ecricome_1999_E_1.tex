\section*{\protect\large Exercice 1}

\subsection*{\textbf{Préliminaire}}

Soit $(x_{n})$ une suite numérique qui vérifie, pour tout entier naturel $n$%
, la relation : 
\begin{equation*}
x_{n+2}=\dfrac{1}{3}x_{n+1}+\dfrac{1}{3}x_{n}
\end{equation*}%
Montrer que : $\lim\limits_{n\rightarrow +\infty }x_{n}=0$\newline
(on donne: $\dfrac{1+\sqrt{13}}{6}=0,77$ à $10^{-2}$ près par excès et $%
\dfrac{1-\sqrt{13}}{6}=-0,44$ à $10^{-2}$ près par défaut).\newline
$a$ et $b$ sont deux réels supérieurs ou égaux à 1.\newline
On étudie la suite numérique $(u_{n})$ définie par : $u_{0}=a\quad u_{1}=b$
\quad et pour tout entier naturel n: 
\begin{equation*}
u_{n+2}=\sqrt{u_{n}}+\sqrt{u_{n+1}}
\end{equation*}

\subsection*{Question 1}

\begin{enumerate}
\item[\textbf{1.a :}] Montrer que, pour tout entier naturel $n$, $u_{n}$ est
bien défini et vérifie : $u_{n}\geqslant 1$.

\item[\textbf{1.b :}] Montrer que la seule limite possible de la suite $%
(u_{n})$ est $4$.

\item[\textbf{1.c :}] Ecrire un programme en Turbo Pascal qui calcule et
affiche la valeur de $u_{n}$ pour des valeurs de $a$ et $b$ réelles supé%
rieures ou égales à $1$ et de $n$ entier supérieur ou égal à $2$, entrées
par l'utilisateur.
\end{enumerate}

\subsection*{Question 2}

On se propose d'établir la convergence de la suite $(u_{n})$ par l'étude
d'une suite auxiliaire $(v_{n})$ définie, pour tout entier naturel $n$ par: 
\begin{equation*}
v_{n}=\dfrac{1}{2}\sqrt{u_{n}}-1
\end{equation*}

\begin{enumerate}
\item[\textbf{2.a :}] Montrer que si $\lim\limits_{n\rightarrow +\infty
}v_{n}=0$ alors $\lim\limits_{n\rightarrow +\infty }u_{n}=4$

\item[\textbf{2.b :}] Vérifier, pour tout entier $n$: $v_{n+2}=\dfrac{%
v_{n+1}+v_{n}}{2(2+v_{n+2})}$.\newline
En déduire que : $|v_{n+2}|\leqslant \dfrac{1}{3}(|v_{n}|+|v_{n+1}|)$.

\item[\textbf{2.c :}] On note $(x_{n})$ la suite définie par : $%
x_{0}=|v_{0}| $, $x_{1}=|v_{1}|$ et pour tout entier naturel $n$ : 
\begin{equation*}
x_{n+2}=\dfrac{1}{3}x_{n+1}+\dfrac{1}{3}x_{n}
\end{equation*}%
Montrer que, pour tout entier naturel $n$, $|v_{n}|\leqslant x_{n}$ et
conclure quant à la convergence de la suite $(u_{n})$.
\end{enumerate}

\section*{\protect\large Exercice 2}

Toutes les matrices de cet exercice sont des éléments de l'ensemble $E$ des
matrices carrées d'ordre 2 à coefficients réels.\newline
On pose : $O=\left( 
\begin{array}{rr}
0 & 0 \\ 
0 & 0%
\end{array}%
\right) $, $I=\left( 
\begin{array}{rr}
1 & 0 \\ 
0 & 1%
\end{array}%
\right) $ et $H=\left\{ M\in E/\exists \alpha \in \mathbb{R}\text{ tel que }%
M=\alpha I\right\} $\newline
et pour toute matrice réelle $M=\left( 
\begin{array}{rr}
a & b \\ 
c & d%
\end{array}%
\right) $, $\tau (M)=a+d$ et $\delta (M)=ad-bc$.

\subsection*{Question 1}

On dit que la suite de matrices $(A_{n})_{n\in \mathbb{N}}$ où $A_{n}=\left( 
\begin{array}{rr}
a_{n} & b_{n} \\ 
c_{n} & d_{n}%
\end{array}%
\right) $ converge vers la matrice $O$ si $(a_{n})$, $(b_{n})$, $(c_{n})$ et 
$(d_{n})$ sont des suites réelles de limite nulle.\newline
Justifier les résultats suivants:

\begin{enumerate}
\item[\textbf{1 a)}] Soit $(A_{n})_{n\in \mathbb{N}}$ et $(B_{n})_{n\in 
\mathbb{N}}$ deux suites de matrices, $\lambda $ un réel et $M$ une matrice:%
\newline
si $(A_{n})_{n\in \mathbb{N}}$ et $(B_{n})_{n\in \mathbb{N}}$ convergent
vers la matrice $O$, alors ($(A_{n}+B_{n})_{n\in \mathbb{N}}$, $(\lambda \
A_{n})_{n\in \mathbb{N}}$, $(M\ A_{n})_{n\in \mathbb{N}}$ et $(A_{n}\
M)_{n\in \mathbb{N}}$ convergent aussi vers $O$.

\item[\textbf{1 b)}] Si $D=\left( 
\begin{array}{rr}
\lambda & 0 \\ 
0 & \mu%
\end{array}%
\right) $ avec $|\lambda |<1$ et $|\mu |<1$, la suite de matrices $%
(D^{n})_{n\in \mathbb{N}}$ converge vers $O$.

\item[\textbf{1 c)}] Si une matrice $A$ est diagonalisable, de valeurs
propres $\lambda $ et $\mu $ tels que $|\lambda |<1$ et $|\mu |<1$, alors la
suite $(D^{n})_{n\in \mathbb{N}}$ converge vers $O$.
\end{enumerate}

\subsection*{Question 2}

Dans toute cette question, $A$ désigne un élément de $E$ tel que $\delta
(A)<0$.\newline
On se propose de montrer qu'une telle matrice est diagonalisable.

\begin{enumerate}
\item[\textbf{2 a)}] Montrer que $A$ n'est pas élément de $H$.

\item[\textbf{2 b)}] Vérifier par le calcul que, pour tout élément $M$ de $E$
on a: 
\begin{equation*}
M^{2}=\tau (M)M-\delta (M)I\qquad (\ast )
\end{equation*}

\item[\textbf{2 c)}] Montrer qu'il existe deux réels distincts $\lambda $ et 
$\mu $ tels que: 
\begin{equation*}
\lambda +\mu =\tau (A)\qquad \text{et }\lambda \mu =\delta (A)
\end{equation*}

\item[\textbf{2 d)}] On pose $M=A-\lambda I$ et $N=A-\mu I$\newline
Montrer que $MN=O$ et en déduire que l'hypothèse \textquotedblleft $M$ est
inversible" conduit à une contradiction.\newline
Montrer de même que $N$ n'est pas inversible.

\item[\textbf{2 e)}] En déduire que $A$ est diagonalisable et qu'il existe
une matrice $P$ de $E$ inversible telle que \newline
$A=P\ D\ P^{-1}$ avec $D=\left( 
\begin{array}{rr}
\lambda & 0 \\ 
0 & \mu%
\end{array}%
\right) $.
\end{enumerate}

\subsection*{Question 3}

On note $U$ l'ouvert de $\mathbb{R}^{2}$ défini par $U=\left] \dfrac{1}{3},{%
\dfrac{2}{3}}\right[ \times ]0,1[$ et $f$ l'application définie sur $U$ par :%
\newline
\begin{equation*}
(x,y)\rightarrow f(x,y)=x^{2}-x+xy^{2}-xy
\end{equation*}

\begin{enumerate}
\item[\textbf{3 a)}] Montrer que $f$ est strictement négative sur $U$.

\item[\textbf{3 b)}] Montrer (en rédigeant soigneusement) que $f$ admet un
unique extremum sur $U$ et que celui-ci est un minimum dont on donnera la
valeur.\newline
En déduire que, pour tout élément $(x,y)$ de $U$: \qquad $-{\dfrac{25}{65}}%
\leqslant f(x,y)<0$.
\end{enumerate}

\subsection*{Question 4}

Soient $a$ et $b$ deux réels tels que $(a,b)$ soit un élément de l'ouvert $U$
défini précédemment.\newline
On pose $Q=\left( 
\begin{array}{cc}
a & b \\ 
a(1-b) & a-1%
\end{array}%
\right) $\newline
On se propose de montrer que la suite de matrices $(Q^{n})_{n\in \mathbb{N}}$
converge vers $O$.

\begin{enumerate}
\item[\textbf{4 a)}] Calculer $\tau (Q)$ et $\delta (Q)$\newline
Vérifier que les résultats de la question 2 s'appliquent pour $A=Q$ et en dé%
duire que $Q$ admet deux valeurs propres distinctes $\lambda $ et $\mu $
telles que: 
\begin{equation*}
-{\dfrac{1}{3}}<\lambda +\mu <{\dfrac{1}{3}}\qquad \text{et}\qquad -{\dfrac{%
25}{64}}\leqslant \lambda \mu \leqslant 0
\end{equation*}

\item[\textbf{4 b)}] Exprimer $\lambda ^{2}+\mu ^{2}$ en fonction de $%
\lambda +\mu $ et $\lambda \mu $ et en déduire que $\lambda ^{2}+\mu ^{2}<1$.%
\newline
Pourquoi peut on affirmer que la suite $(Q^{n})_{n\in \mathbb{N}}$ converge
vers $O$ ?
\end{enumerate}

\section*{Exercice 3}

On modélise la durée de fonctionnement d'un appareil par une variable alé%
atoire réelle $T$ définie sur un certain espace probabilisé $(\Omega ,%
\mathcal{A},P)$ admettant une densité $f$.\newline
On note $F$ sa fonction de répartition et on suppose que $F$ vérifie les
propriétés:

\begin{itemize}
\item $F(t)=0$ pour tout réel $t\leqslant 0$.

\item $F$ est de classe $C^{1}$ et strictement croissante sur $\mathbb{R}%
^{+} $.
\end{itemize}

Sous une hypothèse introduite dans la question 2, on se propose d'expliciter 
$F$ et $f$, puis de calculer l'espérance $E(T)$ de $T$, \textquotedblleft 
\textsl{temps moyen de fonctionnement}".

\subsection*{Question 1}

On rappelle que l'intégrale généralisée $\int\limits_{0}^{+\infty
}e^{-u^{2}}du$ converge et vaut ${\dfrac{1}{2}}\sqrt{\pi }$.\newline
Soit $\alpha $ un réel strictement positif; si $x$ est un élément de $%
\mathbb{R}^{+}$, on pose: 
\begin{equation*}
I(x)=\int\limits_{0}^{x}2u^{2}e^{-u^{2}}du\qquad \text{et}\qquad
J(x)=\int\limits_{0}^{x}t^{2}e^{-(t/\alpha )^{2}}dt
\end{equation*}

\begin{enumerate}
\item[\textbf{1 a)}] A l'aide d'un changement de variable, exprimer, pour
tout élément $x$ de $\mathbb{R}^{+}$,\newline
$J(x)$ en fonction de $I\left( {\dfrac{x}{\alpha }}\right) $.

\item[\textbf{1 b)}] A l'aide d'une intégration par parties, montrer que,
pour tout élément $x$ de $\mathbb{R}^{+}$: 
\begin{equation*}
I(x)=\int\limits_{0}^{x}e^{-u^{2}}du-xe^{-x^{2}}
\end{equation*}

\item[\textbf{1 c)}] En déduire que l'intégrale généralisée $%
\int\limits_{0}^{+\infty }t^{2}e^{-(t/\alpha )^{2}}dt$ converge et vaut ${%
\dfrac{\alpha ^{3}\sqrt{\pi }}{4}}$.
\end{enumerate}

\subsection*{Question 2}

\begin{enumerate}
\item[\textbf{2 a)}] Montrer que, pour tout élément $u$ de $\mathbb{R}^{+}$, 
$F(u)<1$, puis en déduire que $P(T\geqslant u)\neq 0$.

\item[\textbf{2 b)}] Soient $t_{0}$ et $t$ des réels tels que $0\leqslant
t_{0}\leqslant t$.\newline
On pose : $q(t_{0},t)={\dfrac{1}{t-t_{0}}}P(t_{0}\leqslant T\leqslant t\ /\
T\geqslant t_{0})$ (c'est une probabilité conditionnelle).\newline
$q(t_{0},t)$ est \textsl{le taux d'arrêt de fonctionnement entre les
instants $t_{0}$ et $t$}.\newline
On définit ensuite, sous réserve d'existence, le \textsl{taux d'arrêt de
fonctionnement instantané en $t_{0}$} par: 
\begin{equation*}
\tau (t_{0})=\lim_{t\rightarrow t_{0}^{+}}q(t_{0},t)
\end{equation*}%
Exprimer $q(t_{0},t)$ en fonction de $t$, $t_{0}$, $F(t_{0})$ et $F(t)$.%
\newline
En déduire que $\tau (t_{0})$ existe et que $\tau (t_{0})={\dfrac{F^{\prime
}(t_{0})}{1-F(t_{0})}}$\newline
Dans la suite de l'énoncé, on fait l'hypothèse suivante:\newline
il existe un réel $c>0$ tel que, pour tout élément $t$ de $\mathbb{R}^{+}$, $%
\tau (t)=ct$.

\item[\textbf{2 c)}] Montrer que, pour tout élément $t$ de $\mathbb{R}^{+}$,
on a : 
\begin{equation*}
-\ln \left[ 1-F(t)\right] =c{\dfrac{t^{2}}{2}}
\end{equation*}

\item[\textbf{2 d)}] Soit $t$ un élément de $\mathbb{R}^{+}$.\newline
Expliciter $F(t)$, puis montrer, en posant $\alpha =\sqrt{\dfrac{2}{c}}$,
que $f(t)=\dfrac{2}{\alpha ^{2}}te^{-(t/\alpha )^{2}}$.

\item[\textbf{2 e)}] Montrer que $E(T)$ existe et donner sa valeur en
fonction de $c$.
\end{enumerate}
