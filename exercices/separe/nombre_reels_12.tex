\begin{enumerate}
\begin{minipage}[t]{0.45\textwidth}
\item $\sqrt{x+1}=x-1$
\item $\sqrt{x+4}+\sqrt{x+2}\leq 1$
\item $\sqrt{x^2-3}>5x-9$
\item $e^x-1\geq \sqrt{e^{x+1}-e^x-e+1}$
\item $\sqrt{(x+3)(x-1)}\geq 2x-1$
\end{minipage}
\begin{minipage}[t]{0.45\textwidth}
%\item $x\sqrt{x}-4x-\sqrt{x}+4>0$

\item $\sqrt{9^x-1}>3^x-2$
\end{minipage}
\end{enumerate}
\end{exercice}
