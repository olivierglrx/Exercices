
\begin{exercice}  \;
On pose $P(x)=x^2+3x$, $Q(x)=x^2+x+1$, $S(x)=x^2-1$.
\begin{enumerate}
 \item Calculer $P^2(x)$, $P(x)-Q(x)$ et $P^2(x)-Q^2(x)$.
%\item Calculer $P(Q)$, $Q(P)$ et $P(R)-R(P)$.
\item Calculer $P(x+1)$.
\item Calculer $S\circ f(t)$ avec $f: t\mapsto \cos{(t)}$.
\end{enumerate}
\end{exercice}
