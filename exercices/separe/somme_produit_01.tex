
\begin{exercice}  \;
\noindent Soit $n\in\N,\ n\geq 3$. Simplifier les nombres suivants:
$$A=\ddp\frac{7!}{6!},\ \ \ \ B=\ddp\frac{3\times 4!}{(3!)^2},\ \ \ \ 
C=\ddp\frac{n!}{(n-1)!},\ \ \ \ D=\ddp\frac{(n+1)!}{(n-3)!}\ \ \hbox{et}\ \ E=\ddp\frac{(n+1)!}{(n-2)!}+\ddp\frac{n!}{(n-1)!}.$$
\end{exercice}
