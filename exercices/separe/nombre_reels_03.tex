
\begin{exercice} 
R\'esoudre dans $\R$ et selon les valeurs du param\`etre $m\in\R$, les \'equations suivantes:
\begin{enumerate}
\begin{minipage}[t]{0.45\textwidth}
\item $m(x+2)=2m(3x-4)$
\item $(m+1)x+2-m=0$
%\item $mx^2-5m-50=100x$
\item $e^{2x}-2me^x+1=0$
%\item $e^x+e^{-x}=2m$
\end{minipage}
\begin{minipage}[t]{0.45\textwidth}
\item $\ddp\frac{m+3}{x}=\ddp\frac{2m-1}{x-1}$
\item $x-m=\sqrt{x^2+mx}.$
\end{minipage}
\end{enumerate}
\end{exercice}
