\begin{enumerate}
\begin{minipage}[t]{0.45\textwidth}
 \item 
$f(x)=\ddp\frac{1}{\sin^2{x}-3\sin{x}+2}.$  \vsec
\item 
$f(x)=\ddp\frac{\frac{1}{\sqrt{1+\sin{x}}}  +\sqrt{x^2-3x+2}  }{x^2-1}.$
\end{minipage}
\begin{minipage}[t]{0.45\textwidth}
 \item   
$f(x)=\ddp\frac{\sin{(2x)}}{2(1+\cos{x})}.$\vsec
\item

\item 
$f(x)=\ddp\sqrt{\sin^2{x}+\sin{x}-2}.$    
\end{minipage}
\end{enumerate}
