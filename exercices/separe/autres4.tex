\begin{exercice}
Montrer que pour tout $n\in \N$, $$(n+1)! \geq \sum_{k=0}^n k! $$
\footnotesize{Les récurrences c'est bien mais long...}
\end{exercice}



\begin{correction}
Soit $n\in \N$, pour tout $k\in\intent{0,n} $, $k! \leq n! $, donc 
$$\sum_{k=0}^n k! \leq \sum_{k=0}^n n! = (n+1) \times  n ! =(n+1)!$$
\end{correction}
