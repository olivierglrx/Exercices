
\begin{exercice}  \;
R\'esoudre dans $\bC$ les \'equations suivantes et mettre les solutions sous forme exponentielle.
\begin{enumerate}
\begin{minipage}[t]{0.45\textwidth}
\item $z^2=i$
\item $z^3=i$
\item $z^4+4=0$ 
\end{minipage}
\begin{minipage}[t]{0.45\textwidth}
\item $z^2=3-4i$
\item $z^4=j$ (on rappelle que $j=e^{\frac{2i\pi}{3}}$).
\end{minipage}
\end{enumerate}
\end{exercice}
